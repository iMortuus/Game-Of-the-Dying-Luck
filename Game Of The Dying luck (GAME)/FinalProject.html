// utility
void sleep(float seconds){ //delay function per cout
    clock_t startClock = clock();
    float secondsAhead = seconds * CLOCKS_PER_SEC;
    // pause until time
    while(clock() < startClock+secondsAhead);
    return;
}

void createSave(){

			system("cls");

	}
	saveFile.close();
}

void getSave(string& name, string& highscoreName, int& highScore, int& highestRoll){
	ifstream saveFile("highscore.txt");
	if(saveFile.is_open()){
		while(saveFile >> name >> highScore >> highestRoll){
			highscoreName = name;
			highScore = highScore;
			highestRoll = highestRoll;
		}
		saveFile.close();
	}
	
}

void setHighScores(string& name, string& highscoreName, int& numberOfRolls, int& points, int& highscore, int& highestRoll){
	if(points > highscore && numberOfRolls > highestRoll){
		highscoreName = name;
		highscore = points;
		highestRoll = numberOfRolls;
		
		ofstream saveFile;
		saveFile.open ("highscore.txt");
		saveFile << name << " " << points << " " << numberOfRolls;
		saveFile.close();
	}
}

static void handleDoubleEffects(int& points, int color1, int color2, int& numberOfRolls, bool& revive, bool& dead, bool& skipRoll, string& doubleEffect){
	
    int effectNum;
    if(color1 == 1 && color2 == 1) {
    	effectNum = 1;
	}else if(color1 == 2 && color2 == 2){
    	effectNum = 2;
	}else if(color1 == 3 && color2 == 3){
		effectNum = 3;
	}else if(color1 == 4 && color2 == 4){
		effectNum = 4;
	}else if(color1 == 5 && color2 == 5){
		effectNum = 5;
	}else {
		effectNum = 0;
	}
    
    switch(effectNum){ 
        case 1: //double black
            if(!revive){
            	sleep(2);
            	system("color 8");
            	dead = true;
            	doubleEffect = "DEAD!";
			}else{
				revive = false;
				doubleEffect = "\tREVIVED!";
			}
            break;
        case 2: //double blue
        		system("color 1");
            	skipRoll = true;
            	doubleEffect = "ROLL SKIP!";
            break;
        case 3://double yellow
        		system("color 6");
            	points * 2;
            	doubleEffect = "JACKPOT!! POINTS DOUBLED!";
            break;
        case 4://double green
        		system("color 2");
            	revive = true;
            	doubleEffect = "YOU GOT A REVIVE!";
            break;
        case 5://double red
        		system("color 4");
        		numberOfRolls--;
            	doubleEffect = "ROLLS DECREASED!";
            break;
        default:
        		doubleEffect = "";
        	break;
    }
}

void roll(int& dice, string& color, int& pointsUsed, bool& skipRoll, bool& revive){
    dice = rand() % 5 + 1; // randomize the number of the dice from 1 to 5
    if(skipRoll){
    	dice = 0;
	}
    switch(dice){ // deterimines the color of the dice
        case 1:
            color = "Black";
            pointsUsed -= 10;
            break;
        case 2:
            color = "Blue";
            pointsUsed += 2;
            break;
        case 3:
            color = "Yellow";
            pointsUsed += 3;
            break;
        case 4:
            color = "Green";
            pointsUsed += 4;
            break;
        case 5:
            color = "Red";
            pointsUsed += 5;
            break;
        default:
        	color = " ";
        	skipRoll = false;
        	break;
    }
    
    
    if(pointsUsed <= 0 && revive) {
    	pointsUsed = 1;
	}else if(pointsUsed <= 0){
		pointsUsed = 0;
	}
	
    return;
}
